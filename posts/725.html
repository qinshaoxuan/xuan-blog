题目链接：Sliding Window Maximum

题意：给一个数组a[n]，和区间长度k，求出所有区间L , L + k - 1的最大值
 

用单调队列维护最大值，解法同POJ 2823

时间复杂度$$O(n)$$

 

public class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        if (nums.length == 0 || k == 0)    return new int[0];
        int[] ans = new int[nums.length - k + 1];
        int[] q = new int[nums.length];
        int front = 0, rear = -1;
        for (int i = 0; i < nums.length; ++i) {
            while (front <= rear && nums[q[rear]] <= nums[i]) rear–;
            q[++rear] = i;
            if (i - q[front] + 1 > k) front++;
            if (i >= k - 1)    ans[i - k + 1] = nums[q[front]];
        }
        return ans;
    }
}
