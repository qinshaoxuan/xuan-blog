FTP 同步文件 shell 脚本

十一月 21, 2016
  • Shell

最近工作环境换到了mac上，发现修改主题后同步到 FTP 服务器很不方便，于是写了个 shell 脚本来自动上传

因为 FTP 又不支持 shell 命令，而且不能上传整个文件夹，于是用来一个临时 sh 文件来实现递归遍历并上传

使用时要保证 FTP 服务器上文件结构和本地一致

#!/bin/bash
LOCAL_PATH=     # 要上传的文件夹
FTP_PATH=       # 目标文件夹
FTP_SERVER=     # 服务器
USER=           # ftp 用户名
PASSWORD=       # ftp 密码
SH_FILE=        # 临时 shell 文件

cd $LOCAL_PATH
echo “# start upload” > $SH_FILE

function p2f() {
echo $1 >> $SH_FILE
}

function ftpCd() {
p2f “cd $1”
p2f “lcd $1”
}

function ftpCdBack() {
p2f “cd ..”
p2f “lcd ..”
}

function ftpPut() {
p2f “put $1”
}

递归遍历目录

function dfs() {
for i in $(ls); do
echo “$(pwd)/$i”
if [ -d $i ]; then
ftpCd $i
cd $i
dfs
ftpCdBack
cd ..
else
ftpPut $i
fi
done
}

p2f “ftp -n<<EOF”
p2f “open $FTP_SERVER”
p2f “user $USER $PASSWORD”
p2f “binary”
p2f “cd $FTP_PATH”
p2f “lcd $LOCAL_PATH”
p2f “prompt”
dfs
p2f “bye”
p2f “EOF”

sh $SH_FILE
rm $SH_FILE
