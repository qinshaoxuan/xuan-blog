1002 RGCDQ

HDU 5317：http://acm.hdu.edu.cn/showproblem.php?pid=5317

题意：定义(F(x))为x的质因数个数。现有T次询问，每次询问区间([L,R])内的(maxGCD(F(i),F(j))(L\leqslant i < j \leqslant R))

(1\leqslant T \leqslant 10^6, 2\leqslant L < R \leqslant 10^6 )
打表发现(F(x))的最大值只有7，可以先预处理出所有(F(x))的值
在算出所有(dp[i][j]),表示前(i)个(F(x))中有多少个(j)
区间([L,R])内(F(x)=j)的个数(cnt[j]=dp[R][j]-dp[L-1][j])

求(maxGCD)只需要找出最大的(p)满足(cnt[p]\geqslant 2)
注意(p=3)时，为(cnt[3]+cnt[6]\geqslant 2)
(p=2)时，为(cnt[2]+cnt[4]+cnt[6]\geqslant 2)

 

#include <iostream>

#include <cstdio>

#include <cstring>

#include <algorithm>

#include <vector>

#include <cmath>
using namespace std;

#define ll long long

#define N 1000000

int vis[N+10];
int f[N+10];
int dp[N+10][10];

void init()
{
memset(vis,0,sizeof(vis));
for(int i=2;i<=N;++i){
if(!vis[i]){
f[i]++;
vis[i]=1;
for(int j=i+i;j<=N;j+=i){
f[j]++;
vis[j]=1;
}
}
}
for(int i=2;i<=N;++i){
for(int j=1;j<=7;++j){
dp[i][j]=dp[i-1][j];
}
dp[i][f[i]]++;
}
}
int main()
{
init();
int T,l,r;
int cnt[8];
scanf(“%d”,&T);
while(T–){
scanf(“%d%d”,&l,&r);
int ans=1;
memset(cnt,0,sizeof(cnt));
for(int i=1;i<=7;++i)
cnt[i]=dp[r][i]-dp[l-1][i];
for(int i=7;i>=4;–i){
if(cnt[i]>=2){
ans=i;
break;
}
}
if(cnt[3]+cnt[6]>=2)ans=max(ans,3);
if(cnt[2]+cnt[4]+cnt[6]>=2)ans=max(ans,2);
printf(“%d\n”,ans);
}
return 0;
}
 

 

 

 

1004 Painter

HDU 5319：http://acm.hdu.edu.cn/showproblem.php?pid=5319

题意：在一个高度为n的画板上，可以沿斜率为-1的直线涂红色，沿斜率为1的直线涂蓝色，当一个格子被同时涂上红色和蓝色会变成绿色，每个格子只能被同种颜色涂一次，给出涂色后画板的状态，询问最少的涂色次数

( 1\leqslant n \leqslant 50)
模拟

分别处理红色和蓝色的涂色情况，绿色的既当做红色，也当做蓝色

注意下面这种情况：
Ｒ．．．
．Ｒ．．
．．．．
．．．Ｒ
答案是为2

 

#include<cstdio>

#include<algorithm>

#include<iostream>

#include<cmath>

#include<cstring>

#include<queue>

#include<vector>

#include<string>
using namespace std;

#define LL __int64

#define N 55

#define INF 1<<30

int mptr[N][N];
int mptb[N][N];
int n,m;
int vis[N][N];

void change(int x,int y,int dir,int mpt[N][N]) {
if(dir==1) {
for(int i=0; i<=n; ++i) {
if(x+i>n||y+i>m)return;
if(vis[x+i][y+i]||!mpt[x+i][y+i])return;
vis[x+i][y+i]=1;
}
} else {
for(int i=0; i<=n; ++i) {
if(x+i>n||y-i==0)return;
if(vis[x+i][y-i]||!mpt[x+i][y-i])return;
vis[x+i][y-i]=1;
}
}
}

int solve(int mpt[N][N],int dir) {
memset(vis,0,sizeof(vis));
int ret=0;
for(int i=1; i<=n; ++i) {
for(int j=1; j<=m; ++j) {
if(!vis[i][j]&&mpt[i][j]) {
change(i,j,dir,mpt);
ret++;
}
}
}
return ret;
}

int main() {
int T;
scanf(“%d”,&T);
char str[55];
while(T–) {
scanf(“%d”,&n);
memset(mptr,0,sizeof(mptr));
memset(mptb,0,sizeof(mptb));
for(int i=0; i<n; ++i) {
scanf(“%s”,str);
m=strlen(str);
for(int j=0; j<strlen(str); ++j) {
if(str[j]==’.’)mptr[i+1][j+1]=mptb[i+1][j+1]=0;
if(str[j]==’R’)mptr[i+1][j+1]=1;
if(str[j]==’B’)mptb[i+1][j+1]=1;
if(str[j]==’G’)mptr[i+1][j+1]=mptb[i+1][j+1]=1;
}
}

    int ans=solve(mptr,1)+solve(mptb,2);
    printf("%d\n",ans);
}
return 0;

}
 

 

 

 

1008 Solve this interesting problem

HDU 5323：http://acm.hdu.edu.cn/showproblem.php?pid=5323

题意：找到最小的(n)使根节点为([0,n])的线段树存在节点([L,R])

( 1\leqslant L\leqslant R\leqslant 10^9, \frac {L} {R-L+1}\leqslant 2015 )
DFS
设(len=R-L+1)
节点([L,R])的父节点有四种情况：

([L,R+len],[L,R+len-1],[L-len,R],[L-len-1,R])

通过搜索即可找出根节点

 

#include<cstdio>

#include<algorithm>

#include<iostream>

#include<cmath>

#include<cstring>

#include<queue>

#include<vector>

#include<string>
using namespace std;

#define LL __int64

#define N 100005

#define INF 1<<30
LL ans;
LL L,R;

void dfs(LL l,LL r) {
if(l<0)return;
if(l==0) {
ans=min(ans,r);
return;
}
LL len=r-l+1;
if(l<len)return;
dfs(l,r+len);
if(len!=1)
dfs(l,r+len-1);
dfs(l-len,r);
dfs(l-len-1,r);
}

int main() {

while(~scanf("%I64d%I64d",&amp;L,&amp;R)) {
    ans=INF;
    dfs(L,R);
    printf("%I64d\n",ans==INF?-1:ans);
}
return 0;

}
 

 

 

 

1011 Work

HDU 5326：http://acm.hdu.edu.cn/showproblem.php?pid=5326

题意：询问一颗节点数为n的树上子孙节点数为k的节点有多少个

(1\leqslant n \leqslant 10^2 )
DFS，回溯时计算子孙节点

 

#include<cstdio>

#include<algorithm>

#include<iostream>

#include<cmath>

#include<cstring>

#include<queue>

#include<vector>

#include<string>
using namespace std;

#define LL __int64

#define N 100005

#define INF 1<<30

struct node {
int to,next;
} edge[N];
int head[N];
int k;

int n,m;
int cnt[N];
int ans;

void init() {
k=1;
ans=0;
memset(head,0,sizeof(head));
memset(cnt,0,sizeof(cnt));
}

void add(int u,int v) {
edge[k].to=v;
edge[k].next=head[u];
head[u]=k++;
}

int dfs(int u) {
int ret=0;
int v;
for(int i=head[u]; i; i=edge[i].next) {
v=edge[i].to;
ret+=dfs(v);
}
cnt[u]=ret;
return ret+1;
}

int main() {
while(scanf(“%d%d”,&n,&m)!=EOF) {
init();
int a,b;
for(int i=0; i<n-1; ++i) {
scanf(“%d%d”,&a,&b);
add(a,b);
}
dfs(1);
for(int i=1; i<=n; ++i) {
if(cnt[i]==m)
ans++;
}
printf(“%d\n”,ans);
}
return 0;
}
