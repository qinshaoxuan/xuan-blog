神奇的%g

#include<cstdio>

#include<cstring>

char str[100];
int vis[100];
int judge(){
for(int i=0;i<strlen(str);i++){
if(str[i]==’.’) return i;
}
return -1;
}
int main()
{
while(scanf(“%s”,str)!=EOF){
int i,len=strlen(str),pos=judge();
if(pos==-1){
printf(“%s\n”,str);
continue;
}
memset(vis,0,sizeof(vis));
for(i=len-1;i>=pos;i–){
if(str[i]==’0’||str[i]==’.’) vis[i]=1;
else break;
}
for(i=0;i<len;i++){
if(vis[i]==1) break;
printf(“%c”,str[i]);
}
printf(“\n”);
}
return 0;
}//YSF
 

 

 

质因子个数

#include<cstdio>

#include<cstring>

int prime[1005];
int a,b;
void init(){
memset(prime,0,sizeof(prime));
int i,j;
for(i=2;i<=1000;i++){
if(prime[i]!=0) continue;
for(j=2*i;j<=1000;j+=i){
prime[j]++;
}
}
}
int main()
{
init();
while(scanf(“%d%d”,&a,&b)!=EOF){
int sum=0,i;
for(i=a;i<=b;i++) sum+=prime[i];
printf(“%d\n”,sum);
}
return 0;
}//YSF

熊孩子

搬自HDU 5600

0 1 1 0

一对0可以关掉中间的所有1

路线：1234321234

所以偶数个0就YES

升级版 HDU 5601

#include<cstdio>

int main(){
int T;
int n,a,cnt;
scanf(“%d”,&T);
while(T–){
cnt=0;
scanf(“%d”,&n);
for(int i=0;i<n;++i){
scanf(“%d”,&a);
if(!a) cnt++;
}
if(cnt&1)printf(“NO\n”);
else printf(“YES\n”);
}
return 0;
}//xuan
 

 

 

盖伦蹲草丛

#include<cstdio>

#include<cstring>

#include<queue>

#include<algorithm>
using namespace std;

struct node{
int x,y;
};
int n,m,k;
char mpt[105][105];
int d[4][2]={1,0,0,1,0,-1,-1,0};
int BFS(int x,int y){
queue<node>q;
node s,e;
s.x=x;
s.y=y;
int ans=1;
q.push(s);
mpt[x][y]=’.’;
while(!q.empty()){
s=q.front();
q.pop();
for(int i=0;i<4;i++){
e.x=s.x+d[i][0];
e.y=s.y+d[i][1];
if(e.x<1||e.x>n) continue;
if(e.y<1||e.y>m) continue;
if(mpt[e.x][e.y]==’.’) continue;
mpt[e.x][e.y]=’.’;
ans++;
q.push(e);
}
}
return ans;
}
int main()
{
while(scanf(“%d%d%d”,&n,&m,&k)!=EOF){
int i,j;
for(i=1;i<=n;i++) scanf(“%s”,mpt[i]+1);
int max=0;
for(i=1;i<=n;i++){
for(j=1;j<=m;j++){
if(mpt[i][j]==’*’){
int ans=BFS(i,j);
max=(max>ans)?max:ans;
}
}
}
if(max>=k+1) printf(“Demacia Win!\n”);
else printf(“Demacia Lose!\n”);
}
return 0;
}//YSF
 

 

 

放置炸弹

#include<cstdio>

#include<cstring>

int n,m,k,ans;
int vis[20];
void DFS(int index,int num){
if(num==k) {
ans++;
return;
}
if(index>n) return;
DFS(index+1,num);
for(int i=1;i<=m;i++){
if(vis[i]==1) continue;
vis[i]=1;
DFS(index+1,num+1);
vis[i]=0;
}
}
int main()
{
while(scanf(“%d%d%d”,&n,&m,&k)!=EOF){
ans=0;
memset(vis,0,sizeof(vis));
DFS(1,0);
printf(“%d\n”,ans);
}
return 0;
}//YSF
 

 

 

万能钥匙

#include<cstdio>

#include<cstring>

#include<vector>

#include<queue>
using namespace std;

#define INF 0x3f3f3f3f

struct node
{
int data;
int w;
int v;
};
struct node1
{
int step;
int v;
};
vector<node>mpt[1001];
int n,m;
int vis[1001][4];
int dis[1001][4];//表示2-n以不同的方式到1最近的距离
int i,j,k;
void BFS()
{
node1 s,e;
queue<node1>q;
s.step=1;
s.v=1;
vis[s.step][s.v]=1;//标记为在队列中
q.push(s);
while(!q.empty())
{
s=q.front();
q.pop();
vis[s.step][s.v]=0;//标记为不在队列中
int ss=mpt[s.step].size();
for(i=0;i<ss;i++)
{
if(mpt[s.step][i].v==s.v)//判断是否可通过
{
if(dis[s.step][s.v]+mpt[s.step][i].w<dis[mpt[s.step][i].data][s.v%3+1])//是否松弛
{
dis[mpt[s.step][i].data][s.v%3+1]=dis[s.step][s.v]+mpt[s.step][i].w;//松弛
if(vis[mpt[s.step][i].data][s.v%3+1]==0)//判断是否在队列中
{
e.step=mpt[s.step][i].data;
e.v=s.v%3+1;
vis[e.step][e.v]=1;
q.push(e);
}
}
}
}
}
int min=INF;
for(i=1;i<=3;i++)
{
if(dis[n][i]<min)min=dis[n][i];
}
if(min==INF)printf(“-1\n”);
else printf(“%d\n”,min);
}
int main()
{
while(scanf(“%d%d”,&n,&m)!=EOF)
{
memset(vis,0,sizeof(vis));
for(i=1;i<=n;i++)for(j=1;j<=3;j++)dis[i][j]=INF;
dis[1][1]=0;
for(i=1;i<=n;i++)mpt[i].clear();
for(i=1;i<=m;i++)
{
node s;
int x,y;
scanf(“%d%d%d%d”,&x,&y,&s.w,&s.v);
s.data=y;
mpt[x].push_back(s);
s.data=x;
mpt[y].push_back(s);
}
BFS();
}
return 0;
}//YSF
 

 

 

滑雪升级版

#include<cstdio>

#include<cstring>

int mpt[105][105];
int dp[105][105][5];
int d[4][2]={1,0,0,1,-1,0,0,-1};
int n,m;
int DFS(int x,int y,int k){
if(dp[x][y][k]!=0) return dp[x][y][k];
int max=1;
for(int i=0;i<4;i++){
int tx=x+d[i][0];
int ty=y+d[i][1];
if(i==k) continue;
if(tx<1||tx>n) continue;
if(ty<1||ty>m) continue;
if(mpt[tx][ty]>=mpt[x][y]) continue;
int ans=DFS(tx,ty,i)+1;
max=(max>ans)?max:ans;
}
dp[x][y][k]=max;
return max;
}
int main()
{
while(scanf(“%d%d”,&n,&m)!=EOF)
{
memset(dp,0,sizeof(dp));
int i,j,k;
for(i=1;i<=n;i++){
for(j=1;j<=m;j++) scanf(“%d”,&mpt[i][j]);
}
int max=0;
for(i=1;i<=n;i++){
for(j=1;j<=m;j++){
for(k=0;k<4;k++){
dp[i][j][k]=DFS(i,j,k);
max=(max>dp[i][j][k])?max:dp[i][j][k];
}
}
}
printf(“%d\n”,max);
}
return 0;
}//YSF
