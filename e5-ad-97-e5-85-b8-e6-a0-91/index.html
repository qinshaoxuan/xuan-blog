HDU 1251 统计难题
题意：统计出以某个字符串为前缀的单词数量

指针实现

#include<cstdio>

#include<algorithm>

#include<iostream>

#include<cmath>

#include<cstring>

#include<queue>

#include<vector>
using namespace std;

#define N 26

struct node{
node *Next[N];
int v;
node(){
v=0;
for(int i=0;i<N;++i){
Next[i]=NULL;
}
}
};

node *root;

void add(char str){
node p=root;
int k;
for(int i=0;i<strlen(str);++i){
k=str[i]-‘a’;
if(p->Next[k]==NULL){
node *q=new node();
q->v=1;
p->Next[k]=q;
p=p->Next[k];
}else{
p->Next[k]->v++;
p=p->Next[k];
}
}
}

int query(char str){
node p=root;
int k;
for(int i=0;i<strlen(str);++i){
k=str[i]-‘a’;
if(p->Next[k]==NULL)
return 0;
else
p=p->Next[k];
}
return p->v;
}

void del(node root) {
node s = root;
if(s==NULL) return;
for (int i = 0; i < N; i++) {
if (s->Next[i])
del(s->Next[i]);
}
delete s;
}

int main(){
int i;
root=new node();
char str[20];
while(gets(str),str[0]!=’\0’){
add(str);
}
while(scanf(“%s”,str)!=EOF){
printf(“%d\n”,query(str));
}
del(root);
return 0;
}

#include<cstdio>

#include<algorithm>

#include<iostream>

#include<cmath>

#include<cstring>

#include<queue>

#include<vector>
using namespace std;

#define NODE 1000005

#define N 26

int Next[NODE][N]; //数组建树
int v[NODE]; //相同前缀个数
int node; //节点个数

void init(){
node=1;
memset(Next[0],0,sizeof(Next[0]));
}

void add(char *str){
int len=strlen(str);
int cur=0,k;
for(int i=0;i<len;++i){
k=str[i]-‘a’;
if(Next[cur][k]==0){
memset(Next[node],0,sizeof(Next[node]));
v[node]=0;
Next[cur][k]=node++;
}
v[Next[cur][k]]++;
cur=Next[cur][k];
}
}

int query(char *str){
int k,cur=0;
int len=strlen(str);
for(int i=0;i<len;++i){
k=str[i]-‘a’;
if(Next[cur][k])
cur=Next[cur][k];
else
return 0;
}
return v[cur];
}

int main(){
char str[20];
init();
while(1){
gets(str);
if(str[0]==’\0’)break;
add(str);
}
while(scanf(“%s”,str)!=EOF){
printf(“%d\n”,query(str));
}
return 0;
}
