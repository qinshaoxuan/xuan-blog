#define N 1000100

char s[N2+1];//输入时下标从0开始
int p[N2+1];//p[i+i]-1 表示字符串中以第i个字符为中心的回文串长度
//p[i+i+1]-1 表示字符串中以第i个字符与第i+1个字符为中心的回文串长度

void init(char s,int len){
int i;
//在每个相邻的字符之间插入一个分隔符,把奇数长度回文串与偶数长度回文串统一考虑
for(i=len;i>=0;–i){
s[i+i+2]=s[i];
s[i+i+1]=’#’;
}
//插入了len+1个’#’,最终的s长度是1~len+len+1即2len+1,首尾s[0]和s[2len+2]要插入不同的字符
s[0]=’‘;//s[0]=’*’,s[len+len+2]=’\0’,防止在while时p[i]越界

//aba-&gt;*#a#b#a#\0

}

int maxlen=0,id=0;
for(i=1;i<=2len+1;++i){

else
p[i]=1;

    while(s[i-p[i]]==s[i+p[i]]) p[i]++;

    if(id+p[id]&lt;i+p[i])

for(int i=1;i&lt;=2*len+1;++i)
