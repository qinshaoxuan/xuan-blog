//a^b%MOD
LL quickpow(LL a,LL b){
    LL ret=1;
    while(b){
        if(b&1)ret=(ret*a)%MOD;
        a=(a*a)%MOD;
        b>>=1;
    }
    return ret;
}

LL quickmul(LL a,LL b){
    ll ret=0;
    while(b)
    {
        if(b&1){
            ret+=a;
            if(ret>=MOD)ret-=MOD;
        }
        a<<=1;
        if(a>=MOD)a-=MOD;
        b>>=1;
    }
    return ret;
}

矩阵快速幂

解题步骤：
递推公式：$$F(n)=AF(n-1)+BF(n-2)$$

构造一个系数矩阵$$\begin{pmatrix} a & b\ c & d \end{pmatrix}$$(最多与前$$x$$项有关就为$$x\times x$$的系数矩阵）

使$$\begin{pmatrix} F(n-2) & F(n-1) \end{pmatrix} \begin{pmatrix} a & b\ c & d \end{pmatrix}= \begin{pmatrix} F(n-1) & F(n) \end{pmatrix}$$

每次乘以系数矩阵，递推得到一个新的数，并解出未知数（利用递推公式求解）
便可以利用系数矩阵求出$$F(n)$$

$$\begin{pmatrix} F(n-2) & F(n-1) \end{pmatrix} {\begin{pmatrix} a & b\ c & d \end{pmatrix}}^{n-1} = \begin{pmatrix} F(n-1) & F(n) \end{pmatrix}$$

#define N 2

#define MOD 1000000007

//矩阵乘法
//a=ab
void mul(LL a[N][N],LL b[N][N]){
LL c[N][N]={0};
for(int i=0;i<N;++i){
for(int j=0;j<N;++j){
for(int k=0;k<N;++k){
c[i][j]+=(a[i][k]b[k][j]+MOD)%MOD;//防止负数
}
}
}
for(int i=0;i<N;++i){
for(int j=0;j<N;++j){
a[i][j]=(c[i][j]+MOD)%MOD;
}
}
}
//矩阵快速幂
//a=a^b
void matrixpow(LL a[N][N],LL b){
LL e[N][N]={0};//单位矩阵
for(int i=0;i<N;++i)e[i][i]=1;

while(b){
    if(b&amp;1) mul(e,a);
    mul(a,a);
    b&gt;&gt;=1;
}

memcpy(a,e,sizeof(e));//注意不能写sizeof(a)

}
